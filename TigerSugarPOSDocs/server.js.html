/**
 * Express module for building web applications.
 * @type {import('express')}
 */
const express = require('express');
/**
 * Middleware to enable Cross-Origin Resource Sharing (CORS).
 * @type {import('cors')}
 */
const cors = require('cors');
/**
 * PostgreSQL client for Node.js.
 * @type {import('pg').Client}
 */
const { Client } = require('pg');

/**
 * The main Express application instance.
 * @type {import('express').Express}
 */
const app = express();
app.use(cors());
app.use(express.json());

/**
 * PostgreSQL database configuration.
 * @type {Object}
 * @property {string} user - The username for connecting to the database.
 * @property {string} host - The host of the database server.
 * @property {string} database - The name of the database.
 * @property {string} password - The password for connecting to the database.
 * @property {number} port - The port number for the database server.
 */
const dbConfig = {
    user: 'csce315_911_johnammon75',
    host: 'csce-315-db.engr.tamu.edu',
    database: 'csce315331_11b_db',
    password: 'Masonkeegan02!',
    port: 5432
};

/**
 * PostgreSQL client instance for connecting to the database.
 * @type {import('pg').Client}
 */
const client = new Client(dbConfig);

/**
 * Connect to the PostgreSQL database.
 * @param {function(Error)} err - Callback function to handle connection errors.
 */
client.connect(function (err) {
    if (err) {
        console.error('Error connecting to the database:', err);
    } else {
        console.log('Connected to the database!');
    }
});

/**
 * Endpoint for executing a SQL query.
 * @route GET /run-query
 * @function
 * @async
 * @param {import('express').Request} req - Express request object.
 * @param {import('express').Response} res - Express response object.
 * @returns {Promise<void>} - Promise representing the asynchronous operation.
 */
app.get('/run-query', async (req, res) => {
    try {
        const { query } = req.query; // Assuming the query is sent from App.js as a query parameter

        const result = await client.query(query);

        res.json({ result: result.rows });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

/**
 * The port on which the server will listen for incoming requests.
 * @type {number}
 */
const port = process.env.PORT || 8000; // Use default port if available, otherwise, fallback to 8000

/**
 * Start the Express server.
 * @function
 * @param {number} port - The port on which the server will listen for incoming requests.
 * @param {function} callback - Callback function to be executed once the server is running.
 */
app.listen(port, () => {
    console.log(`Server is running on port ${port}.`);
});
